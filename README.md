## Install

```bash
npm install --save weightwedgit
```

## Usage

```jsx
import React, { Component } from 'react'

import Weightwedgit from 'weightwedgit'

class Example extends Component {
  render () {
    const props = {name: 'wight', unit: 'kg', step: 5, value: 80, min: 20, max: 180};
    return (
      <Weightwedgit ...props />
    )
  }
}
```
